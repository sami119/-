using System;
using System.Linq;
using NUnit.Framework;

[TestFixture]
public class AddMethodCount
{
    [Test]
    public void TestMethod()
    {
        Type sut = typeof(MathOperations);

        var methods = sut.GetMethods().Where(n => n.Name == "Add").ToList();

        Assert.IsTrue(methods.Count > 1);
    }
}
